```jsp
<script>
	function no_addnotice() {

		var form = document.getElementById("registerForm");

		form.action = "<c:url value='/notice-register'/>";
		form.submit();

	}

	function no_notice() {

		var form = document.getElementById("registerForm");

		form.action = "<c:url value='/notice-list'/>";
		form.attr("method", "get");
		form.submit();

	}
</script>
```
